version: '3.8'

services:
  web:
    build: ./server
    container_name: server
    command: python -m flask run --host 0.0.0.0 --port 5000 --debug
    ports:
      - '5000:5000'
    depends_on:
      - redis
    volumes:
      - ./server:/app

  redis:
    image: redis:7-alpine

  db:
    image: postgres

  celery_worker:
    build:
      context: ./server
    container_name: celery_worker
    command: celery -A wsgi worker --loglevel INFO
    volumes:
      - ./server:/app
    depends_on:
      - redis
